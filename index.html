// ═══════════════════════════════════════════════════════════════════ // SISTEMA HIS PSICOLOGÍA — Google Apps Script // C.S. Soritor | UPSS: Psicología | MINSA Perú // Servidor: Google Drive + Google Calendar (conexión directa) // ═══════════════════════════════════════════════════════════════════ const FOLDER_NAME = 'HIS Psicología Soritor'; const DATA_FILE = 'hispsy5_data.json'; // ── Servir la aplicación web ────────────────────────────────────── function doGet() { return HtmlService .createHtmlOutputFromFile('Index') .setTitle('Sistema HIS Psicología — C.S. Soritor') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL) .addMetaTag('viewport', 'width=device-width, initial-scale=1.0, maximum-scale=1.0'); } // ── Obtener o crear carpeta en Google Drive ─────────────────────── function getFolder() { const folders = DriveApp.getFoldersByName(FOLDER_NAME); return folders.hasNext() ? folders.next() : DriveApp.createFolder(FOLDER_NAME); } // ── CARGAR datos desde Google Drive ────────────────────────────── function loadData() { try { const folder = getFolder(); const files = folder.getFilesByName(DATA_FILE); if (files.hasNext()) { return files.next().getBlob().getDataAsString('UTF-8'); } return null; } catch(e) { Logger.log('loadData error: ' + e); return null; } } // ── GUARDAR datos en Google Drive ──────────────────────────────── function saveData(jsonStr) { try { const folder = getFolder(); // Guardar archivo principal const files = folder.getFilesByName(DATA_FILE); if (files.hasNext()) { files.next().setContent(jsonStr); } else { folder.createFile(DATA_FILE, jsonStr, MimeType.PLAIN_TEXT); } // Backup automático diario const tz = 'America/Lima'; const hoy = Utilities.formatDate(new Date(), tz, 'yyyy-MM-dd'); const bkName = 'hispsy5_backup_' + hoy + '.json'; const bkFiles = folder.getFilesByName(bkName); if (!bkFiles.hasNext()) { folder.createFile(bkName, jsonStr, MimeType.PLAIN_TEXT); } // Limpiar backups de más de 30 días const limite = new Date(); limite.setDate(limite.getDate() - 30); const allFiles = folder.getFiles(); while (allFiles.hasNext()) { const f = allFiles.next(); const n = f.getName(); if (n.startsWith('hispsy5_backup_') && n.endsWith('.json')) { const fechaStr = n.replace('hispsy5_backup_', '').replace('.json', ''); if (fechaStr < Utilities.formatDate(limite, tz, 'yyyy-MM-dd')) { f.setTrashed(true); } } } return { ok: true, mensaje: 'Guardado en Google Drive ✅' }; } catch(e) { Logger.log('saveData error: ' + e); return { ok: false, error: e.toString() }; } } // ── CREAR evento en Google Calendar ────────────────────────────── function crearCitaCalendar(cita) { try { const cal = CalendarApp.getDefaultCalendar(); const tz = 'America/Lima'; const [y, m, d] = cita.fecha.split('-').map(Number); const [h, min] = (cita.hora || '08:00').split(':').map(Number); const inicio = new Date(y, m - 1, d, h, min, 0); const fin = new Date(y, m - 1, d, h + 1, min, 0); const titulo = (cita.pacNom || cita.pacNombre || 'PACIENTE').toUpperCase(); const desc = [ cita.mot || 'CONSULTA PSICOLÓGICA', 'HC: ' + (cita.pacHC || '—'), 'C.S. Soritor — UPSS Psicología' ].join('\n'); const event = cal.createEvent(titulo, inicio, fin, { description: desc, timeZone: tz }); return { ok: true, gcalId: event.getId(), citaId: cita.id }; } catch(e) { Logger.log('crearCita error: ' + e); return { ok: false, error: e.toString(), citaId: cita.id }; } } // ── ELIMINAR evento de Google Calendar ─────────────────────────── function eliminarCitaCalendar(gcalId) { try { const event = CalendarApp.getEventById(gcalId); if (event) { event.deleteEvent(); return { ok: true }; } return { ok: false, error: 'Evento no encontrado' }; } catch(e) { Logger.log('eliminarCita error: ' + e); return { ok: false, error: e.toString() }; } } // ── SINCRONIZAR lote de citas a Calendar ───────────────────────── function sincronizarCitas(citas) { const resultados = []; citas.forEach(function(cita) { const r = crearCitaCalendar(cita); resultados.push(r); }); return resultados; } // ── LEER eventos de Google Calendar ────────────────────────────── function leerCalendar(fechaMin, fechaMax) { try { const cal = CalendarApp.getDefaultCalendar(); const inicio = new Date(fechaMin + 'T00:00:00-05:00'); const fin = new Date(fechaMax + 'T23:59:59-05:00'); const events = cal.getEvents(inicio, fin); return events.map(function(e) { return { gcalId: e.getId(), titulo: e.getTitle(), inicio: Utilities.formatDate(e.getStartTime(), 'America/Lima', 'yyyy-MM-dd HH:mm'), fin: Utilities.formatDate(e.getEndTime(), 'America/Lima', 'yyyy-MM-dd HH:mm'), descripcion: e.getDescription() }; }); } catch(e) { Logger.log('leerCalendar error: ' + e); return []; } } // ── LISTAR backups disponibles ──────────────────────────────────── function listarBackups() { try { const folder = getFolder(); const files = folder.getFiles(); const backups = []; while (files.hasNext()) { const f = files.next(); const n = f.getName(); if (n.startsWith('hispsy5_backup_') && n.endsWith('.json')) { backups.push({ nombre: n, fecha: n.replace('hispsy5_backup_', '').replace('.json', ''), size: f.getSize() }); } } return backups.sort(function(a, b) { return b.fecha.localeCompare(a.fecha); }); } catch(e) { return []; } } // ── RESTAURAR backup por fecha ──────────────────────────────────── function restaurarBackup(fecha) { try { const folder = getFolder(); const bkName = 'hispsy5_backup_' + fecha + '.json'; const files = folder.getFilesByName(bkName); if (!files.hasNext()) return { ok: false, error: 'Backup no encontrado' }; const data = files.next().getBlob().getDataAsString('UTF-8'); // Restaurar como archivo principal const mainFiles = folder.getFilesByName(DATA_FILE); if (mainFiles.hasNext()) { mainFiles.next().setContent(data); } else { folder.createFile(DATA_FILE, data, MimeType.PLAIN_TEXT); } return { ok: true, data: data }; } catch(e) { return { ok: false, error: e.toString() }; } }
